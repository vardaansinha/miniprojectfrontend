Fibonacci Recursive Algorithm

// Java code for Fibonacci Recursive Algorithm
public class Fibonacci {
    // Recursive function to calculate Fibonacci
    public static int fibonacciRecursive(int n) {
        if (n <= 1) return n;
        return fibonacciRecursive(n - 1) + fibonacciRecursive(n - 2);
    }

    // Main method to test the recursive Fibonacci function
    public static void main(String[] args) {
        // Example: Calculate Fibonacci for n = 5
        int result = fibonacciRecursive(5);
        System.out.println("Fibonacci for n = 5: " + result);
    }
}
